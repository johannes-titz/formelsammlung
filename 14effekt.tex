\subsection*{Mittelwertsunterschiede (modern)}

Für Mittelwertsunterschiede wird die populationsbezogene Effektgröße definiert als
\[
\delta = \frac{\mu_A - \mu_B}{\sigma},
\]

Für Stichproben ergibt sich entsprechend
\[
d = \frac{\bar{x}_A - \bar{x}_B}{\hat{\sigma}},
\]
mit der gepoolten Standardabweichung
\[
\hat{\sigma} =
\sqrt{
\frac{
(n_A - 1)s_A^2 + (n_B - 1)s_B^2
}{
n_A + n_B - 2
}
}.
\]

Da $d$ in endlichen Stichproben leicht nach oben verzerrt ist \parencite{hedges1981}, wird häufig die Korrektur nach Hedges verwendet:
\[
g = J(\mathrm{df})\cdot d
\qquad\text{mit}\qquad
J(\mathrm{df}) \approx 1 - \frac{3}{4\mathrm{df} - 1}.
\]
Für zwei unabhängige Stichproben gilt $\mathrm{df}=n_A+n_B-2$, für Einstichproben- und abhängige Stichproben $\mathrm{df}=n-1$.

Für eine Stichprobe gegen eine vorgegebene Konstante $c$ gilt
\[
d = \frac{\bar{x} - c}{\hat{\sigma}},
\qquad
g = J(\mathrm{df})\cdot d.
\]

Für abhängige Stichproben gilt mit $D$ als Differenzwerten
\[
d = \frac{\bar{D}}{\hat{\sigma}_{D}},
\qquad
g = J(\mathrm{df})\cdot d.
\]


\subsection*{Mittelwertsunterschiede (historisch)}

Nach \textcite[][Formel~2.2.1]{cohen1988} ist die populationsbezogene Effektgröße definiert als
\[
d = \frac{m_A - m_B}{\sigma},
\]
wobei Cohen $m$ für Populationsmittelwerte nutzt. Inhaltlich entspricht dies der modernen Schreibweise mit $\delta$ statt $d$.

Für Stichproben definiert \textcite[][Formel~2.5.1]{cohen1988} den Schätzer
\[
d_s = \frac{\bar{x}_A - \bar{x}_B}{s},
\]
mit
\[
s =
\sqrt{
\frac{
\sum (x_A - \bar{x}_A)^2
+
\sum (x_B - \bar{x}_B)^2
}{
n_A + n_B - 2
}
}.
\]

Diese Formulierung ist algebraisch äquivalent zur modernen Darstellung über die gepoolte Varianz. \textcite{cohen1988} verweist bereits auf die Ergebnisse von \textcite{hedges1981}, nach denen $d_s$ leicht nach oben verzerrt ist.

\subsection*{ANOVA}

US
\begin{align*}
\eta^2 &= \frac{\mathrm{QS}_{\mathrm{zw}}}
               {\mathrm{QS}_{\mathrm{ges}}} \\[0.7em]
\eta^2_{p} &= \frac{\mathrm{QS}_{\mathrm{zw}}}
               {\mathrm{QS}_{\mathrm{zw}} + \mathrm{QS}_{\mathrm{Fehler}}}
\end{align*}

AS
\begin{align*}
\eta^2&=\frac{\mathrm{QS}_{\mathrm{zw}}}{\mathrm{QS}_{\mathrm{ges}}}\\
\eta^2_p&=\frac{\mathrm{QS}_{\mathrm{zw}}}{\mathrm{QS}_{\mathrm{zw}}+\mathrm{QS}_{\mathrm{res}}}\\
\end{align*}
mehrfaktoriell
\begin{align*}
\eta^2=\frac{\mathrm{QS}_{\mathrm{Effekt}}}{\mathrm{QS}_{\mathrm{ges}}}\\
\eta^2_p=\frac{\mathrm{QS}_{\mathrm{Effekt}}}{\mathrm{QS}_{\mathrm{Effekt}}+\mathrm{QS}_{\mathrm{inn}}}\\
\end{align*}

Cohens $f$ (aus $\eta^2$, für Poweranalysen):
\[
f = \sqrt{\frac{\eta^2}{1-\eta^2}}
\]

\subsection*{Kontrast-Analyse}

\begin{align*}
r_{\mathrm{effectsize}}
  &= \frac{\frac{1}{n} \sum_{i=1}^n (x_i - \bar{x})(\lambda_i - \bar{\lambda})}
          {s_x \cdot s_\lambda} \\[0.6em]
r^2_{\mathrm{contrast}}
  &= \frac{\mathrm{QS}_{\mathrm{Kontrast}}}
          {\mathrm{QS}_{\mathrm{Kontrast}} + \mathrm{QS}_{\mathrm{inn}}} \\[0.6em]
r_{\mathrm{alerting}}
  &= \frac{\frac{1}{k} \sum_{i=1}^k (\bar{x}_i - \bar{\bar{x}})(\lambda_i - \bar{\lambda})}
          {s_{\bar{x}} \cdot s_\lambda} \\[0.6em]
r^2_{\mathrm{alerting}}
  &= \frac{\mathrm{QS}_{\mathrm{Kontrast}}}{\mathrm{QS}_{\mathrm{zw}}} \\[0.6em]
d_{\mathrm{contrast}}
  &= \frac{\bar{L}}{\hat{\sigma}_{L}} \\[0.6em]
g_{\mathrm{contrast}}
  &= J(\mathrm{df})\cdot d_{\mathrm{contrast}}
\end{align*}

Zusammenhang der drei Maße:
\[
r_{\mathrm{effectsize}} = \frac{r_{\mathrm{contrast}}}{\sqrt{\,1 - r^2_{\mathrm{contrast}} + r^2_{\mathrm{contrast}}/r^2_{\mathrm{alerting}}\,}}
\]
mit $r_{\mathrm{effectsize}} \le r_{\mathrm{alerting}}$ und $r_{\mathrm{effectsize}} \le r_{\mathrm{contrast}}$ (Gleichheit $r_{\mathrm{effectsize}} = r_{\mathrm{contrast}}$ nur bei $r_{\mathrm{alerting}} = 1$).


\end{multicols}

\begin{boxK}{Effektgrößen aus Signifikanztests}
\begin{multicols}{2}

\subsection*{$t$-Test: Einstichprobenfall}
$$d=\frac{t}{\sqrt{n}}$$
$$g=J(\mathrm{df})\cdot d$$\\

\subsection*{$t$-Test: US}
$$d=t_\mathrm{US}\cdot\sqrt{\frac{1}{n_\mathrm{A}}+\frac{1}{n_\mathrm{B}}}$$
$$g=J(\mathrm{df})\cdot d$$
%wenn $n_\mathrm{A}=n_\mathrm{B}$: 
$$r=\sqrt{\frac{t^2_\mathrm{US}}{t^2_\mathrm{US}+\mathrm{df}}}$$
Letzteres gilt auch für $F$, wenn $\mathrm{df}_{\mathrm{zw}}$ 1 sind, da in diesem Fall $F=t^2$.\\

Vereinfachung, wenn $n_\mathrm{A}=n_\mathrm{B}$: 
$$d=\frac{2\cdot t_\mathrm{US}}{\sqrt{n_\mathrm{A}+n_\mathrm{B}}}$$
$$g=J(\mathrm{df})\cdot d$$


\subsection*{$t$-Test: AS}
$$d=\frac{t_\mathrm{AS}}{\sqrt{n}}$$
$$g=J(\mathrm{df})\cdot d$$
$$r=\sqrt{\frac{t^2_\mathrm{AS}}{t^2_\mathrm{AS}+\mathrm{df}}} \quad (\mathrm{df}=n-1)$$
Hinweis: $d$ und $r$ beziehen sich hier auf die Differenzwerte und sind nicht direkt mit den between-subjects-Maßen (US-Fall, $d\leftrightarrow r$-Umrechnung) vergleichbar.


\subsection*{Kontrastanalyse US}
aus $F$-Test
$$r_{\mathrm{effect size}}=\sqrt{\frac{F_{\mathrm{Kontrast}}}{F_{\mathrm{zw}} \cdot {\mathrm{df}}_{\mathrm{zw}}+{\mathrm{df}}_{\mathrm{inn}}}}$$

$$r_{\mathrm{contrast}} = \sqrt{\frac{F_{\mathrm{Kontrast}}}{F_{\mathrm{Kontrast}}+\mathrm{df_{\mathrm{inn}}}}} = \sqrt{\frac{t^2_{\mathrm{Kontrast}}}{t^2_{\mathrm{Kontrast}}+\mathrm{df}}} $$
\subsection*{Kontrastanalyse AS}
aus t-Test
$$d=\frac{t}{\sqrt{n}}$$
$$g=J(\mathrm{df})\cdot d$$

\subsection*{$\chi^2$-Tests}
$$w=\sqrt{\sum\limits_{i=1}^k\frac{(P_{b,i}-P_{e,\;i})^2}{P_{e,i}}}$$
aus Ergebnis $\chi^2$-Test:
$$w=\sqrt{\frac{\chi^2}{N}}$$
aus 2 dichotomen Variablen
$$w=\sqrt{\frac{\chi^2}{N}}=\phi$$

\subsection*{ANOVA}

US
\begin{align*}
\eta_p^2&=\frac{F \cdot \mathrm{df}_{\mathrm{zw}}}{F \cdot \mathrm{df}_{\mathrm{zw}}+\mathrm{df}_\mathrm{inn}}\\
\end{align*}

Bei einfaktorieller ANOVA gilt: $\eta_p^2=\eta^2$.

\end{multicols}
\end{boxK}

\begin{boxK}{Effektgrößen aus Effektgrößen}
\begin{multicols}{2}

\subsection*{Abstandsmaße aus Abstandsmaßen}
$$d=\frac{g}{J(\mathrm{df})}$$
$$g=J(\mathrm{df})\cdot d$$

\subsection*{Korrelationen aus Abstandsmaßen}
\begin{align*}
r&=\frac{d}{\sqrt{d^2+\frac{\mathrm{df}}{(n_A+n_B)\cdot p\cdot q}}}\\
r&=\frac{\frac{g}{J(\mathrm{df})}}{\sqrt{\left(\frac{g}{J(\mathrm{df})}\right)^2+\frac{\mathrm{df}}{(n_A+n_B)\cdot p\cdot q}}}\\
\end{align*}
Ohne Stichprobengrößen, approximativ mit $p=q=0.5$ und $\mathrm{df}\approx n_A+n_B$ \parencite[z. B.][Formel~2.26]{cohen1988}.
\begin{align*}
r&\approx\frac{d}{\sqrt{d^2+4}}
\end{align*}
\subsection*{Abstandsmaße aus Korrelationen}
\begin{align*}
d&=\frac{r}{\sqrt{1-r^2}}\cdot\sqrt{\frac{(n_A+n_B)\cdot \mathrm{df}}{n_A\cdot n_B}}\\
g&=J(\mathrm{df})\cdot d\\
\end{align*}
Ohne Stichprobengrößen, approximativ mit $p=q=0.5$ und $\mathrm{df}\approx n_A+n_B$.
\begin{align*}
d&\approx\frac{2r}{\sqrt{1-r^2}}
\end{align*}

wobei $$p=\frac{n_A}{n_A+n_B}$$ und $$q=\frac{n_B}{n_A+n_B}$$

\end{multicols}
\end{boxK}
