\subsection*{Kovarianz}
$$\mathrm{cov}(x, y)=\frac{1}{n}\sum\limits_{i=1}^{n}(x_i-\bar{x})(y_i-\bar{y})$$

\subsection*{Korrelation}
Pearson-Korrelationskoeffizient \\
(Produkt-Moment-Korrelation)
$$r=\frac{\mathrm{cov}(x, y)}{s_x s_y}= \frac{1}{n}\sum\limits_{i=1}^{n}{z_{x_i}\cdot z_{y_i}}$$

\subsection*{Partialkorrelation}
$$r_{xy.z}=\frac{r_{xy}- r_{xz}\cdot r_{yz}}{\sqrt{1-r_{xz}^{2}}\cdot\sqrt{1-r_{yz}^{2}}}$$

\subsection*{Phi-Koeffizient (für dichotome Variablen)}
$$\Phi=\frac{a\cdot d-b\cdot c}{\sqrt{(a+b)(c+d)(a+c)(b+d)}}$$

\subsection*{Interrater-Reliabilität (Kappa)}

\begin{align*}
  \kappa &= \frac{p_o - p_e}{1 - p_e}\\
  p_e &= \sum_{i=1}^k p_{i\cdot} \, p_{\cdot i}
\end{align*}

\begin{aligned}
p_o &: \text{beobachtete Übereinstimmung}\\
p_e &: \text{erwartete Übereinstimmung unter Zufall}\\
p_{i\cdot} &:\ \text{die Randwahrscheinlichkeit der Zeile } i\\
p_{\cdot i} &:\ \text{die Randwahrscheinlichkeit der Spalte } i\\
k &:\ \text{die Anzahl der Kategorien}
\end{aligned}
